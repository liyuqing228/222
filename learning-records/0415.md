# Javascript基础知识练习

## 变量的定义

```javascript
var name = "Alice";
let age = 30;
const PI = 3.14;
```

## 打印变量

```javascript
console.log("Name:", name);
console.log("Age:", age);
console.log("PI:", PI);
```

## 变量的类型

```javascript
console.log("Type of name:", typeof name);
console.log("Type of age:", typeof age);
console.log("Type of PI:", typeof PI);
```

## JavaScript中的数据类型

```javascript
let isStudent = true; // boolean
let score = null; // null
let undefinedVar; // undefined

console.log("Is student:", isStudent);
console.log("Score:", score);
console.log("Undefined variable:", undefinedVar);
console.log("Type of isStudent:", typeof isStudent);
console.log("Type of score:", typeof score);
console.log("Type of undefinedVar:", typeof undefinedVar);
```

## 运算符与表达式

### 算术运算符

```javascript
let a = 10;
let b = 5;

console.log("a + b =", a + b); // 加法
console.log("a - b =", a - b); // 减法
console.log("a * b =", a * b); // 乘法
console.log("a / b =", a / b); // 除法
console.log("a % b =", a % b); // 取模
```

### 赋值运算符

```javascript
let c = a; // 赋值
c += b; // 加赋值
console.log("c after += b:", c);
c -= b; // 减赋值
console.log("c after -= b:", c);
c *= b; // 乘赋值
console.log("c after *= b:", c);
c /= b; // 除赋值
console.log("c after /= b:", c);
c %= b; // 取模赋值
console.log("c after %= b:", c);
```

### 逻辑运算符

```javascript
let x = true;
let y = false;

console.log("x && y =", x && y); // 逻辑与
console.log("x || y =", x || y); // 逻辑或
console.log("!x =", !x); // 逻辑非
console.log("!y =", !y); // 逻辑非
```

### 比较运算符

```javascript
console.log("a == b:", a == b); // 等于
console.log("a != b:", a != b); // 不等于
console.log("a === b:", a === b); // 全等于
console.log("a !== b:", a !== b); // 不全等于
console.log("a > b:", a > b); // 大于
console.log("a < b:", a < b); // 小于
console.log("a >= b:", a >= b); // 大于等于
console.log("a <= b:", a <= b); // 小于等于
```

## 流程控制

```javascript
let score2 = 85;

if (score2 >= 90) {
    console.log("Grade: A");
} else if (score2 >= 80) {
    console.log("Grade: B");
} else if (score2 >= 70) {
    console.log("Grade: C");
} else if (score2 >= 60) {
    console.log("Grade: D");
} else {
    console.log("Grade: F");
}
```

## 循环控制

### for循环

```javascript
for (let i = 0; i < 5; i++) {
    console.log("For loop iteration:", i);
}
```

### while循环

```javascript
let j = 0;
while (j < 5) {
    console.log("While loop iteration:", j);
    j++;
}
```

## 函数定义与调用

```javascript
function greet(name) {
    return "Hello, " + name + "!";
}

console.log(greet("Alice"));
```
